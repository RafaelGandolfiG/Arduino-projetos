#include <LiquidCrystal.h>

LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

int trig = 9;
int echo = 10;

void setup() {
  pinMode(trig, OUTPUT);
  pinMode(echo, INPUT);

  lcd.begin(16, 2);
  lcd.setCursor(0, 0);
  lcd.print("Distancia");
}

void loop() {
  long duracao;
  float distancia;

  // Envia pulso
  digitalWrite(trig, LOW);
  delayMicroseconds(2);
  digitalWrite(trig, HIGH);
  delayMicroseconds(10);
  digitalWrite(trig, LOW);

  // Recebe o pulso
  duracao = pulseIn(echo, HIGH);

  // Calcula distância (cm)
  distancia = duracao * 0.034 / 2;

  // Mostra no LCD
  lcd.setCursor(0, 1);
  lcd.print("                "); // limpa linha
  lcd.setCursor(0, 1);
  lcd.print(distancia);
  lcd.print(" cm");

  delay(500);
}
