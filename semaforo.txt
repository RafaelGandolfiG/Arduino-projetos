byte led1=8;
byte led2=9;
byte led3=10;

void setup()
{
  pinMode(led1, OUTPUT);
  pinMode(led2, OUTPUT);
  pinMode(led3, OUTPUT);
}

void loop()
{
  digitalWrite(led1, 1);
  delay(3000); // Wait for 1000 millisecond(s)
  digitalWrite(led1, 0);
  digitalWrite(led3, 1);
  delay(3000);
  digitalWrite(led3, 0);
  digitalWrite(led2, 1);
  delay(1000);
  digitalWrite(led2, 0);
}
